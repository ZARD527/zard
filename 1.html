辗转相除法(欧几里得算法)

循环实现:

#include <iostream>//循环

using namespace std;

// 辗转相除法求最大公约数
int gcd(int a, int b) {
  while (b != 0) {
    int temp = b;
    b = a % b;
    a = temp;
  }
  return a;
}

int main() {
  int num1, num2;
  cin >> num1 >> num2;

  int result = gcd(num1, num2);

  cout << result << endl;

  return 0;
}

递归实现:

#include <iostream>//递归

using namespace std;

int gcd(int a, int b) {
  if (b == 0) {
    return a;
  } else {
    return gcd(b, a % b);
  }
}

int main() {
  int a, b;
  cin >> a >> b;

  int result = gcd(a, b);
  cout << result << endl;

  return 0;
}

三目运算符实现:
